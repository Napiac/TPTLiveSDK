示例代码

应用程序启动时配置直播框架

// Import the necessary SDK classes
#import "TPTLiveFrameworkBuilder.h"
#import "TPTLiveFrameworkBuilderConfig.h"

TPTLiveFrameworkBuilderConfig *config = [[TPTLiveFrameworkBuilderConfig alloc] init];
config.apiHost = @"YourAPIHost";
config.socketIp = @"YourSocketIP";
[TPTLiveFrameworkBuilder appDidLaunch:config];

渠道用户登录

// Import the necessary SDK classes
#import "TPTLiveFrameworkUserEvent.h"
#import "TPTLiveFrameworkUserInParam.h"

// Construct user login parameters
TPTLiveFrameworkUserInParam *param = [[TPTLiveFrameworkUserInParam alloc] init];
param.partner_key = @"YourPartnerKey";
param.partner_user_id = @"YourUserID";

// Perform user login
[TPTLiveFrameworkUserEvent channelUserLoginWithInParam:param handler:^(BOOL success) {
    if (success) {
        NSLog(@"User logged in successfully");
    } else {
        NSLog(@"User login failed");
    }
}];
